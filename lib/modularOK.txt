class Profil {
  final String ime;
  final String mapa;

  Profil({required this.ime, required this.mapa});

  Map<String, dynamic> toJson() => {
        'ime': ime,
        'mapa': mapa,
      };

  static Profil fromJson(Map<String, dynamic> json) {
    return Profil(
      ime: json['ime'],
      mapa: json['mapa'],
    );
  }
}

import 'dart:convert';
import 'package:shared_preferences/shared_preferences.dart';
import '../models/profil.dart';

class ProfilStorageService {
  static const _key = 'profili_json';

  Future<void> spremi(List<Profil> profili) async {
    final prefs = await SharedPreferences.getInstance();
    final json = jsonEncode(profili.map((e) => e.toJson()).toList());
    await prefs.setString(_key, json);
  }

  Future<List<Profil>> ucitaj() async {
    final prefs = await SharedPreferences.getInstance();
    final tekst = prefs.getString(_key);
    if (tekst == null) return [];
    final decoded = jsonDecode(tekst) as List;
    return decoded.map((e) => Profil.fromJson(e)).toList();
  }
}

class _InOutDugmeWidgetState extends State<InOutDugmeWidget> {
  bool jeOut = true;

  void _toggle() {

      jeOut = !jeOut;

  @override
  Widget build(BuildContext context) {

      child: Text(jeOut ? 'OUT' : 'IN'),
    );
  }
